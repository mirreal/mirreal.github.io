function createRandomArray(num) {
  var list = [];

  while(1) {
    var randomNumber = Math.floor( Math.random() * num );
    if ( list.indexOf(randomNumber) === -1 ) {
      list.push(randomNumber)
    }

    if (list.length === num) break;
  }

  return list;
}
